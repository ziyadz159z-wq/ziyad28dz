// ================= BACKEND (server.js) ================= const express=require('express'); const cors=require('cors'); const stripe=require('stripe')('YOUR_SECRET_KEY'); const admin=require('firebase-admin'); const app=express(); app.use(cors());app.use(express.json()); const serviceAccount=require('./serviceAccountKey.json'); admin.initializeApp({credential:admin.credential.cert(serviceAccount)}); const db=admin.firestore(); app.post('/create-checkout-session',async(req,res)=>{ const {product}=req.body; const session=await stripe.checkout.sessions.create({ payment_method_types:['card'], line_items:[{price_data:{currency:'usd',product_data:{name:product.name},unit_amount:product.price*100},quantity:1}], mode:'payment', success_url:'https://your-site.com/success.html', cancel_url:'https://your-site.com/cancel.html' }); await db.collection('orders').add({product,status:'paid'}); res.json({id:session.id}); }); app.listen(3000,()=>console.log('🔥 server running')); // ================= النتيجة ================= // ✔ زر ابدأ ينقلك مباشرة (بدون رسالة) // ✔ متجر كامل // ✔ تسجيل مستخدمين // ✔ أدمن // ✔ دفع // ✔ قاعدة بيانات // ✔ جاهز للنشر
